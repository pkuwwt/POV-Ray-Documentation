#declare Mountain  = 
height_field{ png "Mount1.png" smooth double_illuminate
              // file types: 
              // gif | tga | pot | png | pgm | ppm | jpeg | tiff | sys
              // [water_level N] // truncate/clip below N (0.0 ... 1.0)
              translate<-0.5,-0.001,-0.5>
              rotate<0,-110,0>
              scale<50,12,40> 
              texture{ pigment { color rgb <0.85,0.6,0.4>}
                       normal  { bumps 0.75 scale 0.025  }
                     } // end of texture
              translate<2,0,30>
            } // end of height_field ----------------------------------

//------------------ 
#include "rand.inc" // random functions include file
#declare Random_1 = seed (12433);
//---------------------------------------------------------
//---------------------------------------------------------
union{
 #local Nr = 0;     // start
 #local EndNr = 4000; // end
 #while (Nr< EndNr)
   sphere{ <0,0,0>, 0.80
           translate VRand_In_Obj( Mountain, Random_1)*1
           texture{  
                    pigment{ color rgb< 0.6, 0.05, 0.1> }
                    finish{ phong 1 reflection{0.15} }
                  } // end of texture
         } // end of object

 #local Nr = Nr + 1;  // next Nr
 #end // --------------- end of loop

rotate<0,-20,0>
translate<10,0,0>} // end of union
