Motivi regulari: tiling - Nuovo con POV-Ray 3.7
tiling = la pavimentazione del piano x-z con 24 motivi classici.
            (lunghezza originale di un quadrato = 1)
Esempio: 

    
box{ <0,0,0>,<1,1,1>
     texture {
        pigment{
           tiling 5 // 1~24 Pattern_Number
              color_map{
                [ 0.0 color rgb<1,1,1>]
                [ 1.0 color rgb<0,0,0>]
                } // end color_map
           scale 0.15
         } // end pigment
        finish { phong 1}
      } // end of texture
   } // end of box

'tiling' e 'frequency'



 Un effetto interessante insieme con frequency:

  

  
#declare Pigment_1 =
   pigment{
      tiling 6 // 1~24 Pattern_Number
      color_map{
         [0.0 color rgb<0,0,0>]
         [0.4 color rgb<1,0,0>]
         [1.0 color rgb<0,0,0>]
       } // end color_map
      rotate<-90,0,0>
      frequency 5
      scale<0.05,0.25,1>*0.5
    } // end pigment
//------------------------------------
torus{ 1, 0.333
  texture { uv_mapping
     pigment{ uv_mapping Pigment_1 }
     finish { phong 0.1}
  } // end of texture
translate<0.5,0.75,0.5>
} // end of torus --------------------
