» material_map { ... } « - Mapping de textures par des images de motifs

On obtient une texture qui utilise des textures différentes selon les indices de couleur d'une image.
// Avec palette de couleur (ex. 256 : image gif couleur)
// Dans le plan xy de <0,0,0> à <1,1,0> .(Projection parallèle vers z)
Sample » material_map « :
 
box {<0,0,0>,< 1,1,1> scale 2 rotate<0,-20,0> translate<-0.5,0,0>
  texture{ material_map {gif "plasma3.gif" // fichier image
          // (accepté : iff, gif, tga, png, sys)
          map_type 0
          // Projection type:
          // 0=planar, 1=spherical, 2=cylindrical, 5=torus
          interpolate 2
          // Interpolation type: 0=none, 1=linear, 2=bilinear,
          // 4=normalized distance
          // [once] // optional -       // texture pour
          texture { EMBWood1 scale 0.3 }// l'index de couleur n°0
          texture { EMBWood1 scale 0.3 }// l'index de couleur n°1
          texture { EMBWood1 scale 0.3 }// l'index de couleur n°2
          texture { Polished_Chrome    }// l'index de couleur n°3
          texture { Polished_Chrome    }// l'index de couleur n°4
          texture { Polished_Chrome    }// l'index de couleur n°5
          texture { pigment {Pine_Wood}}// l'index de couleur n°6
          texture { pigment {Pine_Wood}}// l'index de couleur n°7
          texture { pigment {Pine_Wood}}// l'index de couleur n°8
          texture { pigment {Pine_Wood}}// l'index de couleur n°9
          texture { pigment {Pine_Wood}}// l'index de couleur n°10
          texture { pigment {Pine_Wood}}// l'index de couleur n°11
          texture { pigment {Pine_Wood}}// l'index de couleur n°12
          texture { pigment {Pine_Wood}}// l'index de couleur n°13
          texture { pigment {Pine_Wood}}// l'index de couleur n°14
          texture { pigment {Jade }    }// l'index de couleur n°15
          // etc. etc. etc. ---
         } // fin de material_map
       scale 1
      } // fin de texture -----------------------------------------
 } // fin de sphere ----------------------------
