                                        		Cette forme est définie dans mon include file "shapes3.inc" .
Half_Hollowed_Rounded_Cylinder1

Fichier exemple comme fichier ".txt" ou ".pov"
Syntaxe générale : 
object{ Half_Hollowed_Rounded_Cylinder2
                    (  Length_total,
                       R_out,
                       R_Corner,
                       R_Border,
                       Border_Scale,
                       Merge_On
                    ) //----------------
         texture{ ... }
      } // end of object ---------------
Length_total = longueur total,
R_out >0, rayon externe,
R_Corner = rayon d'angle,
R_Border = rayon du bord, >=0, < (corner radius)/2,
Border_Scale y ( >0 ), 0 = sans bords arrondis !
Merge_On : 0 = union, 1 = merge.
Half_Hollowed_Rounded_Cylinder2

Fichier exemple comme fichier ".txt" ou ".pov"
Syntaxe générale : 

 object{ Half_Hollowed_Rounded_Cylinder2
                    (  Length_total,
                       R_out,
                       R_Corner,
                       R_Border,
                       Border_Scale,
                       Merge_On
                    ) //----------------
         texture{ ... }
      } // end of object ---------------
Length_total = total_Lenght from end to end,
R_out >0, rayon externe,
R_Corner = rayon d'angle,
R_Border rayon du bord, >=0, < (corner radius)/2,
Border_Scale y ( >0 ), 0 = sans bords arrondis !
Merge_On 0 = union, 1 = merge.
Exemple Half_Hollowed_Rounded_Cylinder1 : 
#include "shapes3.inc"
object { Half_Hollowed_Rounded_Cylinder1(
            3.2,// Length_total,
            0.9,// outer radius
            0.2,// border radius
            1, //  border scale y (>=0),
            0  //  Merge_On,
        ) // ------------------------------
        texture{ pigment{ color rgb<1,1,1>}
                 finish { phong 0.1}
               } // end texture
        scale <1,1,1>
        rotate <0,0,0>
        translate < 0, 0.9, 0>
      } // end of object ------------------
