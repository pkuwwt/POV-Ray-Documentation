isosurface{ //-------------------------
  function{  f_checker( x*1.2,y*1.19,z*1.2)*0.08
          }
  contained_by{box{<-2,-3,-2>,<2,3,2>}}
  threshold 0.05
  max_gradient 200

  texture{
    pigment{ color rgb<1,1,1>*1.2}
            finish { phong 1 }
  } // end of texture
  scale <1,1,1>*0.35
  rotate <0,0,0>
  translate <0,0,0>
} // end of isosurface ----------------
