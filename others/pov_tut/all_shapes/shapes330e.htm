// -----------------------------------
object {
 // Round_Box(A, B, WireRadius, Merge)
 Round_Box(<0,0,0>,<1,0.5,2>, 0.125, 0)
 texture{
   pigment{ color rgb<0.5,0.5,1.0>}
   finish { phong 1}
 } // end of texture
 scale<1,1,1>
 rotate<0,0,0>
 translate<0,0,0>
} //----------------------------------
 
// ----------------------------------
object {

   finish {  phong 1}
 } // end of texture

 rotate<0,0,0>

} //---------------------------------
