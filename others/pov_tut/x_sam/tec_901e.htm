Rocket 1
Example for the using of the "union"-statement
Objects: sphere, cone, prism.
Methods: union. 

//------------ rocket ---------------------------
union{
 //------ the body ------------------------------
cone{ <0,0,0>,0.5,<0,1.5,0>,0.25
      texture{Chrome_Metal}}
 //------- the controll surfaces ----------------
prism { -0.01 ,0.01 , 4
        <-1.00, 0.00>,
        < 1.00, 0.00>,
        < 0.00, 1.50>,
        <-1.00, 0.00>
        rotate<-90,0,0> translate<0,0,0>
        texture {pigment{color White }
                 finish {diffuse 0.9 phong 1.0}}
  } // -------------- xy-direction ---------------
prism { -0.01 ,0.01 , 4
        <-1.00, 0.00>,
        < 1.00, 0.00>,
        < 0.00, 1.50>,
        <-1.00, 0.00>
        rotate<-90,90,0>
        translate<0,0,0>
        texture {pigment{color White}
                 finish {diffuse 0.9 phong 1.0}}
  } // -------------- zy-direction ---------------
 //------- the top of the rocket -----------------
sphere{ <0,0,0>,0.25
        scale <1,1.5,1>
        translate<0,1.5,0>
        texture{Chrome_Metal}}
 //-----------------------------------------------
 //----------------- ready for take off:
rotate<0,0,0> translate<0,0,0>
} //end of union
//-------------------------------------------- end
