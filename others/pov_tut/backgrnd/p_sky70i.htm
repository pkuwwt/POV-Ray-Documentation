Un cielo al tramonto:
Per simulare un cielo al tramonto possiamo usare un color_map con un gradiente dal blu scuro al rosso brillante y sul sky_sphere. Possiamo girare questo pigmento color_map contro l'asse x per simulare un cielo rossiccio nella parte occidentale (qui: z direzione)
// sky ----------------------------------
sky_sphere{
 pigment{ gradient <0,1,0>
          color_map{
     [0.00 color rgb<0.24,0.32,1> *0.3]
     [0.23 color rgb<0.16,0.32,0.9> *0.9]
     [0.37 color rgb<1,0.1,0> ]
     [0.52 color rgb<1,0.2,0> ]
     [0.70 color rgb<0.36,0.32,1> *0.7 ]
     [0.80 color rgb<0.14,0.32,1> *0.5 ]
     [1.00 color rgb<0.24,0.32,1> *0.3 ]
                   } // end color_map
     scale 2
     rotate <-20,0,0>
     translate <0,0.7,0>
 } // end pigment
} // end sky_sphere ---------------------
