Nuvole massicche con "media":

Con media si può creare nuvole con structure massicche anche bene per l'effetto di un volo attraverso le nuvole. Qui una "density" nuvoloso è dolcemente ristretto con una "density" di tipo "boxed". Per più di dettagli si può vedere mio tutorial Design del Interno dei Oggetti trasparenti con POV-Ray. 
// nuvole con media --------------------
box{ <-1,-1,-1>,<1,1,1>
  texture{
    pigment{ rgbf 1} // color Clear
  } // end of texture
  interior{
   media{
    method 3
    //intervals 2
    samples 10,10
    absorption 1
    emission 0.5
    scattering{ 0.8,<1,1,1>*0.5}
    density{ bozo
             color_map {
              [0.00 rgb 0]
              [0.50 rgb 0.01]
              [0.65 rgb 0.1]
              [0.75 rgb 0.5]
              [1.00 rgb 0.2]
             } // end color_map
             turbulence 0.85
             scale  0.75
             translate<1, 0.75,2>
           } // end density
    density{  boxed // or: spherical
              color_map {
               [0.0 rgb 0]    // border
               [0.1 rgb 0.05]
               [1.0 rgb 1]    // center
              } // end color_map
             scale <1,1,1>*1
           } // end density
   } // end media
  } // end interior
 hollow
 scale<1,0.5, 1> translate<0,0.5,0>
 //-------------------------------------
scale 15 translate<0,02,30>
}//-------------------------------------
